overmiliterisation_strategy_training_war = {
	enable = {
		has_war = yes 
		set_temp_variable = { THIS.num_battalions_max_war = THIS.num_battalions_max }	#Per 1000 Max Battalions
		multiply_temp_variable = {
			THIS.num_battalions_max_war = 1.05												#Will train up to 1050
		}
		
		if = { limit = { surrender_progress > 0.1 }
			multiply_temp_variable = {
				THIS.num_battalions_max_war = 1.1											#Will train up to 1155
			}
		}
		if = { limit = { surrender_progress > 0.25 }
			multiply_temp_variable = {
				THIS.num_battalions_max_war = 1.2											#Will train up to 1386
			}
		}
		if = { limit = { surrender_progress > 0.5 }
			multiply_temp_variable = {
				THIS.num_battalions_max_war = 1.3											#Will train up to 1802
			}
		}
		if = { limit = { threat > 0.5 }
			multiply_temp_variable = {
				THIS.num_battalions_max_war = 1.15											#Will train up to 2072
			}
		}
		if = { limit = { has_war_with_major = yes }
			multiply_temp_variable = {
				THIS.num_battalions_max_war = 1.1											#Will train up to 2279
			}
		}
		
		round_temp_variable = THIS.num_battalions_max_war
		
		check_variable = {
			var = THIS.num_battalions
			value = THIS.num_battalions_max_war
			compare = less_than_or_equals
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
	type = build_army
		id = garrison
		value = 800
	}
	ai_strategy = {
		type = build_army
		id = suppression
		value = 500
	}
	ai_strategy = {
		type = build_army
		id = infantry
		value = 1500
	}
	ai_strategy = {
		type = build_army
		id = support
		value = 1000
	}
	ai_strategy = {
		type = build_army
		id = anti_air
		value = 1000
	}
	ai_strategy = {
	type = build_army
		id = anti_tank
		value = 1000
	}
	ai_strategy = {
		type = build_army
		id = artillery
		value = 1000
	}

		value = 1000
	}
}
