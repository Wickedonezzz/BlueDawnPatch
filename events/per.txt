					}			
			}

			SYR = {
					add_popularity = {
							ideology = paternal_autocracy
							popularity = -0.25
					}
					add_popularity = {
							ideology = socialism
							popularity = 0.25
					}
			}		
	
		}
	}
	
	#big kurdistan#
	option = {
		name = per.1.a
		KUR = {
		transfer_state = 676

		}

		factor = 55
		}
	}
	
	#give them the iraqi kurdistan#
	option = {
		name = per.1.b
		KUR = {
		transfer_state = 676
		}
		ai_chance = {
		factor = 65
		}
	}
	
	#no changes #

		name = per.1.c
		ai_chance = {
		factor = 45
		}
	}

#lebanese civil war#
country_event = {
	id = per.2
	title = per.2.t
	desc = per.2.d
	picture = GFX_report_event_IRQ_kurds
	
	trigger = {
		original_tag = PER
		has_country_flag = PER_gamer_win
		date > 2005.05.05
	}
	
	fire_only_once = yes

	#lets go#
	option = {
		name = per.2.a
		HZB = {
		transfer_state = 953
		
			declare_war_on = {
				target = LEB
				type = annex_everything
			}
			
		}
		
		PER = {add_to_faction = HZB}
		
		country_event = {id = per.3 days = 156}		
		ai_chance = {
		factor = 55
		}
	}
}
