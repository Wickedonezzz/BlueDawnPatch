#ethiopia war#
country_event = {
	id = foreign_policy.1
	title = foreign_policy.1.t
	desc = foreign_policy.1.d
	
	is_triggered_only = yes
	
	#back ethiopia
	option = {
		name = foreign_policy.1.a	
		set_country_flag = backed_ethiopia
		custom_effect_tooltip = foreign_policy_explanation
	}
	
	#back Eritrea
	option = {
		name = foreign_policy.1.b
		set_country_flag = backed_eritrea	
		custom_effect_tooltip = foreign_policy_explanation		
	}
}

#the war results arrived#
country_event = {
	id = foreign_policy_award.1
	title = foreign_policy_award.1.t
	desc = foreign_policy_award.1.d
	
	trigger = {
		ETH = {has_war = no}
		OR = {
			has_country_flag = backed_ethiopia
			has_country_flag = backed_eritrea
		}
	
	}
	
	fire_only_once = yes
	
	#ethiopia won
	option = {
		name = foreign_policy_award.1.a	
		trigger = {
			country_exists = ETH
		}
		if = {
			limit = {has_country_flag = backed_ethiopia}
			add_offsite_building = { type = industrial_complex level = 1 }
			add_offsite_building = { type = arms_factory level = 1 }			
			add_political_power = 75
			ETH = {
				add_opinion_modifier = { target = FROM modifier = supported_us }
                add_opinion_modifier = { target = FROM modifier = supported_us_trade }
			}	
			FROM = {
				add_opinion_modifier = { target = ETH modifier = supported_us }
				add_to_variable = {generic_fp_won = 1}
			}
		}
		else_if = {
			add_political_power = -75
			ETH = {
				add_opinion_modifier = { target = FROM modifier = supported_the_enemy }
                add_opinion_modifier = { target = FROM modifier = supported_the_enemy_trade }
			}
			FROM = {
				add_to_variable = {generic_fp_lost = 1}
			}
		}
	}
	
	#Eritrea won
	option = {
		name = foreign_policy_award.1.b
		trigger = {
			country_exists = ERI
		}
		if = {
			limit = {has_country_flag = backed_eritrea}
			add_offsite_building = { type = industrial_complex level = 1 }
			add_offsite_building = { type = arms_factory level = 1 }	
			add_political_power = 75
			ERI = {
				add_opinion_modifier = { target = FROM modifier = supported_us }
                add_opinion_modifier = { target = FROM modifier = supported_us_trade }
			}	
			FROM = {
				add_opinion_modifier = { target = ERI modifier = supported_us }
				add_to_variable = {generic_fp_won = 1}
			}			
		}	
		else_if = {
			add_political_power = -75
			ERI = {
				add_opinion_modifier = { target = FROM modifier = supported_the_enemy }
                add_opinion_modifier = { target = FROM modifier = supported_the_enemy_trade }
			}
			FROM = {
				add_to_variable = {generic_fp_lost = 1}
			}			
		}
